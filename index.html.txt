const REQ_CROSSES = 5; 

let gs=null, raf=null, keys={};

function show(id){
  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('on'));
  document.getElementById(id).classList.add('on');

function goTitle(){cancelAnimationFrame(raf);show('sTitle');}

const CV=document.getElementById('cv');
const CTX=CV.getContext('2d');

function startGame(){
  CV.width=window.innerWidth;
  CV.height=window.innerHeight;
  const W=CV.width,H=CV.height;
  const RT=Math.round(H*.32), RB=Math.round(H*.60);
  const LANES=4, LH=(RB-RT)/LANES;

  const animal=ANIMALS[Math.floor(Math.random()*ANIMALS.length)];

  gs={
    animal, W, H, RT, RB, LH,
    px:W/2, py:RB+H*.1,
    vx:0, vy:0,
    lives:3, crosses:0,
    t0:Date.now(),
    phase:'bottom',
    paused:false,
    invTimer:0,
    hitFlash:0, safeGlow:0,
    cars:[], spawnT:0,
    pulses:[],
    particles:[],
    sense:{h:0,s:0,v:0},
    tick:0,
    darkAmt:0.2,
    dead:false,
  };

  updateLiveDots();
  document.getElementById('hudRight').textContent='× 0';
  keys={};
  show('sGame');
  cancelAnimationFrame(raf);
  raf=requestAnimationFrame(loop);
}

function loop(){
  update();
  draw();
  if(gs && !gs.dead) {
    raf=requestAnimationFrame(loop);
  }
}

function update(){
  if(!gs||gs.dead)return;
  const{animal:a,W,H,RT,RB}=gs;
  gs.tick++;

  gs.darkAmt=Math.min(0.96, gs.darkAmt + 0.00008 + gs.crosses*0.00005);

  if(gs.hitFlash>0)gs.hitFlash--;
  if(gs.safeGlow>0)gs.safeGlow--;
  if(gs.invTimer>0)gs.invTimer--;

  if(gs.paused)return;

  gs.spawnT++;
  const diff=1+gs.crosses*0.18;
  const iv=Math.max(18, 45-gs.crosses*4);
  if(gs.spawnT>=iv){gs.spawnT=0;spawnCar(diff);}

  for(let i=gs.cars.length-1; i>=0; i--){
    const c=gs.cars[i];
    c.x+=c.vx;
    if(gs.tick%24===i%24){
      gs.pulses.push({x:c.x,y:c.y,r:4,maxR:Math.max(W,H)*.65,alpha:.55,spd:4.5});
    }
    if(c.x>W+150||c.x<-150)gs.cars.splice(i,1);
  }

  for(let i=gs.pulses.length-1; i>=0; i--){
    const p=gs.pulses[i];
    p.r+=p.spd;
    p.alpha=.55*(1-p.r/p.maxR);
    if(p.r>=p.maxR)gs.pulses.splice(i,1);
  }

  const sp=a.sp;
  if(keys['ArrowLeft']||keys['mL']) gs.vx-=sp*.45;
  else if(keys['ArrowRight']||keys['mR']) gs.vx+=sp*.45;
  else gs.vx*=.70;
  
  if(keys['ArrowUp']||keys['mU']) gs.vy-=sp*.50;
  else if(keys['ArrowDown']||keys['mD']) gs.vy+=sp*.50; 
  else gs.vy*=.70;
  
  gs.vx=Math.max(-sp,Math.min(sp,gs.vx));
  gs.vy=Math.max(-sp,Math.min(sp,gs.vy));
  gs.px+=gs.vx; gs.py+=gs.vy;
  
  gs.px=Math.max(14,Math.min(W-14,gs.px));
  gs.py=Math.max(14,Math.min(H-14,gs.py));

  if((gs.phase==='bottom'||gs.phase==='returning')&&gs.vy<0&&gs.py<RB+6)
    gs.phase='crossing';
    
  if(gs.phase==='crossing'&&gs.py<RT-12){
    gs.phase='top';
    gs.crosses++;
    document.getElementById('hudRight').textContent='× '+gs.crosses;
    
    if(gs.crosses >= REQ_CROSSES) {
      cancelAnimationFrame(raf);
      setTimeout(()=>showSurvive(), 400);
      return;
    }
    
    gs.safeGlow=60;
    gs.darkAmt=Math.max(0.15, gs.darkAmt-.12);
    addPart(gs.px,gs.py,'#406040',14);
    setTimeout(()=>{if(gs&&gs.phase==='top'){gs.phase='returning';}},800);
  }
  if(gs.phase==='returning'&&gs.py>RB+H*.05)
    gs.phase='bottom';

  if(gs.phase==='crossing' && gs.invTimer===0){
    for(const c of gs.cars){
      if(Math.abs(c.x-gs.px)<c.hw+10&&Math.abs(c.y-gs.py)<c.hh+8){
        hitPlayer(c);break;
      }
    }
  }

  updateSense();

  for(let i=gs.particles.length-1; i>=0; i--){
    const p=gs.particles[i];
    p.x+=p.vx;p.y+=p.vy;p.vy+=.07;p.life--;
    if(p.life<=0)gs.particles.splice(i,1);
  }
}

function spawnCar(diff){
  const{RT,RB,LH,W}=gs;
  const lane=Math.floor(Math.random()*4);
  const y=RT+LH*lane+LH/2;
  const right=Math.random()>.5;
  const spd=(2.4+Math.random()*2.8)*diff*(right?1:-1);
  const w=50+Math.random()*30,h=20;
  gs.cars.push({x:right?-90:W+90,y,vx:spd,hw:w/2,hh:h/2,w,h,hl:Math.random()>.15});
}

function hitPlayer(car){
  gs.lives--;
  gs.hitFlash=25;
  gs.invTimer=55;
  addPart(gs.px,gs.py,'#6a1a1a',18);
  updateLiveDots();

  if(gs.lives<=0){
    gs.dead=true;
    const fl=document.getElementById('flash');
    fl.style.transition='none';
    fl.style.background='#fff';fl.style.opacity='1';
    setTimeout(()=>{fl.style.transition='opacity .4s';fl.style.opacity='0';},160);
    cancelAnimationFrame(raf);
    setTimeout(()=>showDeath(),800);
  } else {
    gs.py=gs.RB+gs.H*.1;
    gs.phase='bottom';gs.vy=0;
  }
}

function updateSense(){
  const{animal:a,px,py,cars,W,H,RT,RB}=gs;
  
  let minD = Math.hypot(W,H);
  let horizMinD = W; 
  
  for(const c of cars){
    const d = Math.hypot(c.x-px,c.y-py);
    if(d < minD) minD = d;
    
    const hDist = Math.abs(c.x - px);
    if(hDist < horizMinD) horizMinD = hDist;
  }
  
  const hT = horizMinD < 420*a.he ? (1 - horizMinD/(420*a.he)) * a.he : 0;
  const sT = minD < 220*a.sm ? Math.pow(1 - minD/(220*a.sm), 1.3) * a.sm : 0;
  const isOnRoad = (py >= RT && py <= RB);
  const vT = (isOnRoad && minD < 180) ? (1 - minD/180) : 0;
  
  gs.sense.h += (hT - gs.sense.h) * .12;
  gs.sense.s += (sT - gs.sense.s) * .08;
  gs.sense.v += (vT - gs.sense.v) * .25;
  
  const pct = v => Math.round(Math.min(1,v)*100)+'%';
  document.getElementById('sfH').style.width=pct(gs.sense.h);
  document.getElementById('sfS').style.width=pct(gs.sense.s);
  document.getElementById('sfV').style.width=pct(gs.sense.v);
  
  document.getElementById('sfH').style.background=gs.sense.h>.7?'#8a4a10':'#5a4010';

function addPart(x,y,color,n){
  for(let i=0; i<n; i++){
    const a=Math.random()*Math.PI*2,s=Math.random()*3+.5;
    gs.particles.push({x,y,vx:Math.cos(a)*s,vy:Math.sin(a)*s,color,life:24+Math.random()*14,r:Math.random()*2+.5});
  }
}

function draw(){
  if(!gs)return;
  const{W,H,RT,RB,LH,px,py,darkAmt,tick}=gs;
  CTX.clearRect(0,0,W,H);

  const sky=CTX.createLinearGradient(0,0,0,RT);
  sky.addColorStop(0,'#02060a');sky.addColorStop(1,'#05100a');
  CTX.fillStyle=sky;CTX.fillRect(0,0,W,RT);

  for(let i=0; i<24; i++){
    const sx=(i*157.3)%W, sy=(i*61.7)%(RT-10)+5;
    const tw=.3+.5*Math.abs(Math.sin(tick*.012+i*.7));
    CTX.globalAlpha=tw*.5;
    CTX.fillStyle='#fff';
    CTX.fillRect(sx,sy,1.2,1.2);
  }
  CTX.globalAlpha=1;

  drawForest(CTX,0,RT,W,false);

  CTX.fillStyle='#141414';CTX.fillRect(0,RT,W,RB-RT);
  CTX.setLineDash([14,13]);CTX.lineWidth=1;CTX.strokeStyle='rgba(200,150,30,.15)';
  for(let i=1; i<4; i++){
    const y=RT+LH*i;
    CTX.beginPath();CTX.moveTo(0,y);CTX.lineTo(W,y);CTX.stroke();
  }
  CTX.setLineDash([]);
  CTX.strokeStyle='rgba(255,255,255,.15)';CTX.lineWidth=1.5;
  CTX.beginPath();CTX.moveTo(0,RT);CTX.lineTo(W,RT);CTX.stroke();
  CTX.beginPath();CTX.moveTo(0,RB);CTX.lineTo(W,RB);CTX.stroke();

  const bg=CTX.createLinearGradient(0,RB,0,H);
  bg.addColorStop(0,'#0e1e0e');bg.addColorStop(1,'#060e06');
  CTX.fillStyle=bg;CTX.fillRect(0,RB,W,H-RB);
  drawForest(CTX,0,RB,W,true);

  for(const p of gs.pulses){
    CTX.globalAlpha=p.alpha*.65;
    CTX.strokeStyle='#c8820a';
    CTX.lineWidth=1;
    CTX.beginPath();CTX.arc(p.x,p.y,p.r,0,Math.PI*2);CTX.stroke();
  }
  CTX.globalAlpha=1;

  gs.cars.forEach(c=>drawCar(CTX,c));

  gs.particles.forEach(p=>{
    CTX.globalAlpha=p.life/40;
    CTX.fillStyle=p.color;
    CTX.beginPath();CTX.arc(p.x,p.y,p.r,0,Math.PI*2);CTX.fill();
  });
  CTX.globalAlpha=1;

  const visR=Math.max(40, 70+gs.animal.si*110-darkAmt*65);
  const fog=CTX.createRadialGradient(px,py,visR*.3,px,py,visR);
  fog.addColorStop(0,`rgba(0,0,0,${darkAmt*.15})`);
  fog.addColorStop(.6,`rgba(0,0,0,${darkAmt*.75})`);
  fog.addColorStop(1,`rgba(0,0,0,${Math.min(.98,darkAmt+.05)})`);
  CTX.fillStyle=fog;CTX.fillRect(0,0,W,H);

  CTX.save();
  CTX.fillStyle=`rgba(0,0,0,${Math.min(.99,darkAmt+.06)})`;
  CTX.beginPath();
  CTX.rect(0,0,W,H);
  CTX.arc(px,py,visR,0,Math.PI*2,true);
  CTX.fill('evenodd');
  CTX.restore();

  const sv=gs.sense.v;
  if(sv>0.05){
    const ag=CTX.createRadialGradient(px,py,2,px,py,20+sv*55);
    ag.addColorStop(0,`rgba(90,150,200,${sv*.25})`);
    ag.addColorStop(1,'transparent');
    CTX.fillStyle=ag;CTX.beginPath();CTX.arc(px,py,20+sv*55,0,Math.PI*2);CTX.fill();
  }

  drawPlayerDot(CTX,px,py);

  if(gs.hitFlash>0){CTX.fillStyle=`rgba(180,0,0,${gs.hitFlash/25*.45})`;CTX.fillRect(0,0,W,H);}
  if(gs.safeGlow>0){CTX.fillStyle=`rgba(40,80,40,${gs.safeGlow/60*.2})`;CTX.fillRect(0,0,W,H);}
}

function drawForest(ctx,x,y,w,below){
  const cols=['#080f08','#0e1a0e','#0a140a','#101808'];
  const n=Math.ceil(w/16)+2;
  for(let i=0; i<n; i++){
    const tx=x+i*16+4;
    const th=24+Math.sin(i*2.1+(below?3.5:0))*8;
    const ty=below?y:y-th;
    ctx.fillStyle=cols[i%4];
    ctx.beginPath();ctx.moveTo(tx,ty);ctx.lineTo(tx-7,ty+th);ctx.lineTo(tx+7,ty+th);ctx.closePath();ctx.fill();
  }
}

function drawCar(ctx,c){
  ctx.save();ctx.translate(c.x,c.y);
  ctx.fillStyle='rgba(25,25,25,.95)';
  ctx.beginPath();ctx.roundRect(-c.hw,-c.hh,c.w,c.h,4);ctx.fill();
  if(c.hl){
    const hx=c.vx>0?c.hw:-c.hw;
    const dir=c.vx>0?1:-1;
    const gd=ctx.createRadialGradient(hx+dir*25,0,1,hx+dir*25,0,90);
    gd.addColorStop(0,'rgba(255,240,170,.75)');
    gd.addColorStop(.4,'rgba(255,220,110,.25)');
    gd.addColorStop(1,'transparent');
    ctx.fillStyle=gd;ctx.beginPath();ctx.arc(hx+dir*25,0,90,0,Math.PI*2);ctx.fill();
  }
  ctx.restore();
}

function drawPlayerDot(ctx,x,y){
  if(gs.invTimer>0&&Math.floor(gs.invTimer/5)%2===0)return;
  ctx.save();ctx.translate(x,y);
  ctx.fillStyle='rgba(0,0,0,.3)';
  ctx.beginPath();ctx.ellipse(0,5,6,2,0,0,Math.PI*2);ctx.fill();
  ctx.fillStyle='rgba(210,230,200,.95)';
  ctx.beginPath();ctx.arc(0,0,5,0,Math.PI*2);ctx.fill();
  ctx.restore();
}
